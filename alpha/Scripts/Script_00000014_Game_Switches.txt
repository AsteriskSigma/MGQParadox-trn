#==============================================================================
# ■ Game_Switches
#------------------------------------------------------------------------------
# It is a class dealing with switches. It is a wrapper around the built-in class Array. This class
# Instances of$game_switches It is referred to.
#==============================================================================

class Game_Switches
  #--------------------------------------------------------------------------
  # ● Object initialization
  #--------------------------------------------------------------------------
  def initialize
    @data = []
  end
  #--------------------------------------------------------------------------
  # ● Obtaining the switch
  #--------------------------------------------------------------------------
  def [](switch_id)
    @data[switch_id] || false
  end
  #--------------------------------------------------------------------------
  # ● Switch setting
  #     value : ON (true) / OFF (false)
  #--------------------------------------------------------------------------
  def []=(switch_id, value)
    @data[switch_id] = value
    on_change
  end
  #--------------------------------------------------------------------------
  # ● Processing when setting the switch
  #--------------------------------------------------------------------------
  def on_change
    $game_map.need_refresh = true
  end
end
