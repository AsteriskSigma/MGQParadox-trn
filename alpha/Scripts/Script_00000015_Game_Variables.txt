#==============================================================================
# ■ Game_Variables
#------------------------------------------------------------------------------
# Class to deal with variables. It is a wrapper around the built-in class Array. In this class
# The instance is$game_variables It is referred to.
#==============================================================================

class Game_Variables
  #--------------------------------------------------------------------------
  # ● Object initialization
  #--------------------------------------------------------------------------
  def initialize
    @data = []
  end
  #--------------------------------------------------------------------------
  # ● Obtaining variables
  #--------------------------------------------------------------------------
  def [](variable_id)
    @data[variable_id] || 0
  end
  #--------------------------------------------------------------------------
  # ● Setting variables
  #--------------------------------------------------------------------------
  def []=(variable_id, value)
    @data[variable_id] = value
    on_change
  end
  #--------------------------------------------------------------------------
  # ● Processing when setting variables
  #--------------------------------------------------------------------------
  def on_change
    $game_map.need_refresh = true
  end
end
