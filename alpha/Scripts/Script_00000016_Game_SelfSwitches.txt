#==============================================================================
# ■ Game_SelfSwitches
#------------------------------------------------------------------------------
# It is a class dealing with self-switch. It is a wrapper for the built-in class Hash. This
# An instance of Ras$game_self_switches It is referred to.
#==============================================================================

class Game_SelfSwitches
  #--------------------------------------------------------------------------
  # ● Object initialization
  #--------------------------------------------------------------------------
  def initialize
    @data = {}
  end
  #--------------------------------------------------------------------------
  # ● Acquisition of self switch
  #--------------------------------------------------------------------------
  def [](key)
    @data[key] == true
  end
  #--------------------------------------------------------------------------
  # ● Self-switch setting
  #     value : ON (true) / OFF (false)
  #--------------------------------------------------------------------------
  def []=(key, value)
    @data[key] = value
    on_change
  end
  #--------------------------------------------------------------------------
  # ● Processing when setting the self switch
  #--------------------------------------------------------------------------
  def on_change
    $game_map.need_refresh = true
  end
end
