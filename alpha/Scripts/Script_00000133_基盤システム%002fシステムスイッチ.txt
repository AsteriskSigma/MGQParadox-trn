#==============================================================================
# ■ Game_SystemSwitches
#==============================================================================
class Game_SystemSwitches
  #--------------------------------------------------------------------------
  # ● Object initialization
  #--------------------------------------------------------------------------
  def initialize
    @data = {}
  end
  #--------------------------------------------------------------------------
  # ● Acquire system switch
  #--------------------------------------------------------------------------
  def [](key)
    @data[key] == true
  end
  #--------------------------------------------------------------------------
  # ● System switch setting
  #     value : ON (true) / OFF (false)
  #--------------------------------------------------------------------------
  def []=(key, value)
    @data[key] = value
    on_change
  end
  #--------------------------------------------------------------------------
  # ● Processing when setting system switch
  #--------------------------------------------------------------------------
  def on_change
    $game_map.need_refresh = true
  end
end

#==============================================================================
# ■ Game_Interpreter
#==============================================================================
class Game_Interpreter
  #--------------------------------------------------------------------------
  # ● Acquire system switch
  #--------------------------------------------------------------------------  
  def ssw(key)
    return $game_system_switches[key]
  end
  #--------------------------------------------------------------------------
  # ● System switch on
  #--------------------------------------------------------------------------  
  def on_ssw(key)
    $game_system_switches[key] = true
  end
  #--------------------------------------------------------------------------
  # ● System switch off
  #--------------------------------------------------------------------------  
  def off_ssw(key)
    $game_system_switches[key] = false
  end
  #--------------------------------------------------------------------------
  # ● Flip system switch
  #--------------------------------------------------------------------------  
  def flip_ssw(key)
    $game_system_switches[key] = !$game_system_switches[key]
  end
end
